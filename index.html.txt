const api = (path, opts = {}) => fetch(`${SUPABASE_URL}/rest/v1${path}`, {
  headers: {
    'apikey': SUPABASE_KEY,

// ── 상태 ──
let stocks = [];
let currentTicker = null;
let currentFilter = '전체';

// ── 초기 로드 ──
async function init() {
  await loadStocks();
}

async function loadStocks() {
  const data = await api('/stocks?order=created_at.desc&select=*');
  stocks = Array.isArray(data) ? data : [];
  renderStockList();
  document.getElementById('last-updated').textContent = new Date().toLocaleTimeString('ko-KR', {hour:'2-digit', minute:'2-digit'}) + ' 기준';
}

function renderStockList() {
  const list = document.getElementById('stock-list');
  const filtered = currentFilter === '전체' ? stocks : stocks.filter(s => s.status === currentFilter);

  if (!filtered.length) {
    list.innerHTML = '<div class="loading">종목이 없습니다</div>';
    return;
  }

  list.innerHTML = filtered.map(s => `
    <div class="stock-item ${s.ticker === currentTicker ? 'active' : ''}" onclick="selectStock('${s.ticker}')">
      <div class="stock-item-name">${s.name}</div>
      <div class="stock-item-meta">
        <span>${s.ticker}</span>
        <span>${s.sector || ''}</span>
        <span class="badge badge-${s.status}">${s.status}</span>
      </div>
    </div>
  `).join('');
}

function setFilter(filter, btn) {
  currentFilter = filter;
  document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
  btn.classList.add('active');
  renderStockList();
}

// ── 종목 선택 ──
async function selectStock(ticker) {
  currentTicker = ticker;
  renderStockList();

  document.getElementById('empty-state').style.display = 'none';
  document.getElementById('stock-detail').style.display = 'block';
  document.getElementById('stock-detail').innerHTML = '<div class="loading">불러오는 중...</div>';

  const stock = stocks.find(s => s.ticker === ticker);
  const [financials, notes] = await Promise.all([
    api(`/financials?ticker=eq.${ticker}&order=date.desc&limit=1`),
    api(`/analysis_notes?ticker=eq.${ticker}&order=date.desc`)
  ]);

  const fin = financials?.[0] || {};
  renderDetail(stock, fin, notes || []);
}

function fmt(v, unit='') {
  if (v == null) return '-';
  if (unit === '억') return Number(v).toLocaleString('ko-KR') + '억';
  if (unit === '원') return Number(v).toLocaleString('ko-KR') + '원';
  return Number(v).toLocaleString('ko-KR') + unit;
}

// ── 메모 저장 ──
async function saveMemo(ticker) {
  const bull = document.getElementById('memo-bull').value.trim();
  const bear = document.getElementById('memo-bear').value.trim();
  const target = document.getElementById('memo-target').value;
  const memo = document.getElementById('memo-text').value.trim();

  if (!bull && !bear && !memo) { showToast('내용을 입력해주세요', true); return; }

  await api('/analysis_notes', {
    method: 'POST',
    prefer: 'return=minimal',
    body: JSON.stringify({
      ticker,
      date: new Date().toISOString().split('T')[0],
      bull_case: bull || null,
      bear_case: bear || null,
      target_price: target ? Number(target) : null,
      memo: memo || null
    })
  });

  showToast('✅ 저장됐어요!');
  selectStock(ticker);
}

function openAddModal() { document.getElementById('add-modal').classList.add('open'); }
function closeAddModal() { document.getElementById('add-modal').classList.remove('open'); }

async function addStock() {
  const ticker = document.getElementById('input-ticker').value.trim();
  const name   = document.getElementById('input-name').value.trim();
  const sector = document.getElementById('input-sector').value;
  const market = document.getElementById('input-market').value;
  const status = document.getElementById('input-status').value;

  if (!ticker || !name) { showToast('종목코드와 이름을 입력해주세요', true); return; }

  await api('/stocks', {
    method: 'POST',
    prefer: 'return=minimal',
    body: JSON.stringify({ ticker, name, sector, market, status })
  });

  closeAddModal();
  showToast(`✅ ${name} 추가됐어요!`);
  await loadStocks();
  selectStock(ticker);
}

function showToast(msg, isError = false) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.style.background = isError ? 'var(--red)' : 'var(--green)';
  t.classList.add('show');
  setTimeout(() => t.classList.remove('show'), 2500);
}
